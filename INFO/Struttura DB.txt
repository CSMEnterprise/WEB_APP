CREATE TABLE Admin (
  id_admin INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(255) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  livello_sicurezza INT NOT NULL DEFAULT 1,
  data_creazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CHECK (livello_sicurezza BETWEEN 1 AND 5)
);

CREATE TABLE Utente_Registrato (
  id_utente INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(255) NOT NULL UNIQUE,
  username VARCHAR(100) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  nome VARCHAR(100),
  telefono VARCHAR(20),
  indirizzo TEXT,
  propic VARCHAR(255),
  stato_ban BOOLEAN NOT NULL DEFAULT FALSE,
  data_registrazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Account_Business (
  id_acc_business INT AUTO_INCREMENT PRIMARY KEY,
  id_utente INT NOT NULL UNIQUE,
  p_iva VARCHAR(20) NOT NULL UNIQUE,
  nome_azienda VARCHAR(255) NOT NULL,
  logo VARCHAR(255),
  descrizione TEXT,
  indirizzo TEXT,
  telefono VARCHAR(20),
  email_aziendale VARCHAR(255) NOT NULL UNIQUE,
  link_social VARCHAR(255),
  verificato BOOLEAN NOT NULL DEFAULT FALSE,
  data_registrazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  id_admin_verifica INT,
  data_verifica TIMESTAMP NULL,

  FOREIGN KEY (id_utente) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_admin_verifica) REFERENCES Admin(id_admin)
    ON DELETE SET NULL,

  CHECK (
    (verificato = FALSE AND data_verifica IS NULL)
    OR
    (verificato = TRUE AND data_verifica IS NOT NULL)
  )
);

CREATE TABLE Categoria (
  id_categoria INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100) NOT NULL UNIQUE,
  id_padre INT DEFAULT NULL,

  FOREIGN KEY (id_padre) REFERENCES Categoria(id_categoria)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

CREATE TABLE Annuncio (
  id_annuncio INT AUTO_INCREMENT PRIMARY KEY,
  id_utente INT NULL,
  id_business INT NULL,
  id_categoria INT NOT NULL,
  titolo VARCHAR(255) NOT NULL,
  descrizione TEXT,
  stato_conservazione ENUM('Nuovo','Ottimo','Buono','Discreto','Da restaurare') NOT NULL,
  prezzo DECIMAL(10,2) NOT NULL,
  modalita_consegna ENUM('Spedizione','Ritiro_a_mano','Entrambi') NOT NULL,
  stato ENUM('attivo','venduto') NOT NULL DEFAULT 'attivo',
  data_creazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_scadenza TIMESTAMP NULL,

  FOREIGN KEY (id_utente) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_business) REFERENCES Account_Business(id_acc_business)
    ON DELETE CASCADE,

  FOREIGN KEY (id_categoria) REFERENCES Categoria(id_categoria)
    ON DELETE RESTRICT,

  CHECK (prezzo >= 0),

  CHECK (
    (id_utente IS NOT NULL AND id_business IS NULL)
    OR
    (id_utente IS NULL AND id_business IS NOT NULL)
  )
);

CREATE TABLE Immagine (
  id_immagine INT AUTO_INCREMENT PRIMARY KEY,
  id_annuncio INT NOT NULL,
  url VARCHAR(500) NOT NULL,
  ordine TINYINT NOT NULL DEFAULT 0,

  FOREIGN KEY (id_annuncio) REFERENCES Annuncio(id_annuncio)
    ON DELETE CASCADE,

  UNIQUE (id_annuncio, ordine),
  CHECK (ordine >= 0)
);

CREATE TABLE Pagamento (
  id_pagamento INT AUTO_INCREMENT PRIMARY KEY,
  id_annuncio INT NOT NULL,
  id_acquirente INT NOT NULL,
  importo_totale DECIMAL(10,2) NOT NULL,
  stato ENUM('In_attesa','Completato','Annullato','Rimborsato')
    NOT NULL DEFAULT 'In_attesa',
  paypal_transaction_id VARCHAR(255) UNIQUE,
  data TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (id_annuncio)
    REFERENCES Annuncio(id_annuncio)
    ON DELETE RESTRICT,

  FOREIGN KEY (id_acquirente)
    REFERENCES Utente_Registrato(id_utente)
    ON DELETE RESTRICT,

  CHECK (importo_totale >= 0)
);

CREATE TABLE Feedback (
  id_feedback INT AUTO_INCREMENT PRIMARY KEY,
  id_autore INT NOT NULL,
  id_pagamento INT NOT NULL UNIQUE,
  valutazione TINYINT NOT NULL,
  commento TEXT,
  data_feedback TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (id_autore) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_pagamento) REFERENCES Pagamento(id_pagamento)
    ON DELETE CASCADE,

  CHECK (valutazione BETWEEN 1 AND 5)
);

CREATE TABLE Segnalazione (
  id_segnalazione INT AUTO_INCREMENT PRIMARY KEY,

  id_segnalante INT NOT NULL,

  id_annuncio INT NULL,
  id_utente_segnalato INT NULL,
  id_business INT NULL,
  id_feedback INT NULL,

  tipologia ENUM('Spam','Truffa','Contenuto_inappropriato','Altro') NOT NULL,
  descrizione TEXT,
  stato ENUM('Aperta','In_revisione','Risolta') NOT NULL DEFAULT 'Aperta',
  data_segnalazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  id_admin INT NULL,
  data_risoluzione TIMESTAMP NULL,

  FOREIGN KEY (id_segnalante) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_utente_segnalato) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_admin) REFERENCES Admin(id_admin)
    ON DELETE SET NULL,

  FOREIGN KEY (id_feedback) REFERENCES Feedback(id_feedback)
    ON DELETE CASCADE,

  FOREIGN KEY (id_annuncio) REFERENCES Annuncio(id_annuncio)
    ON DELETE CASCADE,

  FOREIGN KEY (id_business) REFERENCES Account_Business(id_acc_business)
    ON DELETE CASCADE,

  CHECK (
    (id_annuncio IS NOT NULL AND id_utente_segnalato IS NULL AND id_business IS NULL AND id_feedback IS NULL)
    OR
    (id_annuncio IS NULL AND id_utente_segnalato IS NOT NULL AND id_business IS NULL AND id_feedback IS NULL)
    OR
    (id_annuncio IS NULL AND id_utente_segnalato IS NULL AND id_business IS NOT NULL AND id_feedback IS NULL)
    OR
    (id_annuncio IS NULL AND id_utente_segnalato IS NULL AND id_business IS NULL AND id_feedback IS NOT NULL)
  )
);

CREATE TABLE Modera (
  id_moderazione INT AUTO_INCREMENT PRIMARY KEY,
  id_admin INT NOT NULL,

  id_utente INT NULL,
  id_feedback INT NULL,
  id_annuncio INT NULL,
  id_business INT NULL,

  azione_compiuta TEXT NOT NULL,
  data_azione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (id_admin) REFERENCES Admin(id_admin)
    ON DELETE RESTRICT,

  FOREIGN KEY (id_utente) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_feedback) REFERENCES Feedback(id_feedback)
    ON DELETE CASCADE,

  FOREIGN KEY (id_annuncio) REFERENCES Annuncio(id_annuncio)
    ON DELETE CASCADE,

  FOREIGN KEY (id_business) REFERENCES Account_Business(id_acc_business)
    ON DELETE CASCADE,

  CHECK (
    (id_utente IS NOT NULL AND id_feedback IS NULL AND id_annuncio IS NULL AND id_business IS NULL)
    OR
    (id_utente IS NULL AND id_feedback IS NOT NULL AND id_annuncio IS NULL AND id_business IS NULL)
    OR
    (id_utente IS NULL AND id_feedback IS NULL AND id_annuncio IS NOT NULL AND id_business IS NULL)
    OR
    (id_utente IS NULL AND id_feedback IS NULL AND id_annuncio IS NULL AND id_business IS NOT NULL)
  )
);

CREATE TABLE Preferito (
  id_utente INT NOT NULL,
  id_annuncio INT NOT NULL,
  data_aggiunta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  PRIMARY KEY (id_utente, id_annuncio),

  FOREIGN KEY (id_utente) REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE,

  FOREIGN KEY (id_annuncio) REFERENCES Annuncio(id_annuncio)
    ON DELETE CASCADE
);

CREATE TABLE Carrello (
  id_carrello INT AUTO_INCREMENT PRIMARY KEY,
  id_utente INT NOT NULL UNIQUE,
  data_creazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_aggiornamento TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  FOREIGN KEY (id_utente)
    REFERENCES Utente_Registrato(id_utente)
    ON DELETE CASCADE
);

CREATE TABLE Elemento_Carrello (
  id_elemento_carrello INT AUTO_INCREMENT PRIMARY KEY,
  id_carrello INT NOT NULL,
  id_annuncio INT NOT NULL,
  data_aggiunta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (id_carrello)
    REFERENCES Carrello(id_carrello)
    ON DELETE CASCADE,

  FOREIGN KEY (id_annuncio)
    REFERENCES Annuncio(id_annuncio)
    ON DELETE CASCADE,

  UNIQUE (id_carrello, id_annuncio)
);
